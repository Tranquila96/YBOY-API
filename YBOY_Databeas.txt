create table role_(
role_id int GENERATED BY DEFAULT AS IDENTITY primary key,
role_name varchar2(50)
)

create table user_login(
user_id int GENERATED BY DEFAULT AS IDENTITY primary key,
Fname varchar2(50),
Lname varchar2(50),
user_email varchar2(50),
user_password varchar2(50),
phone varchar2(20),
user_imagepath varchar2(1000),


role_id int,
CONSTRAINT user_role_id_fk FOREIGN KEY (role_id) REFERENCES role_(role_id)
)

create table website(
web_id int GENERATED BY DEFAULT AS IDENTITY primary key,
web_name varchar2(50),
web_image_logo_path varchar2(1000),
web_image_packground_path varchar2(1000),
address varchar2(100),
phone varchar2(20),
email varchar2(50),

card_id int,
CONSTRAINT website_card_fk FOREIGN KEY (card_id) REFERENCES card(card_id)

)


create table about_us(
about_us_id int GENERATED BY DEFAULT AS IDENTITY primary key,
info varchar2(2000),
image_path varchar2(1000), 
web_id int,

CONSTRAINT web_about_fk FOREIGN KEY (web_id) REFERENCES website(web_id)
)

create table contact_us(
contact_us_id int GENERATED BY DEFAULT AS IDENTITY primary key,
Fname varchar2(50),
Lname varchar2(50),
phone varchar2(20),
message varchar2(1000),
web_id int,

CONSTRAINT web_contact_fk FOREIGN KEY (web_id) REFERENCES website(web_id)
)

create table testimonials(
testimonial_id int GENERATED BY DEFAULT AS IDENTITY primary key,
Fname varchar2(50),
Lname varchar2(50),
image_path varchar2(1000),
feedback_mess VARCHAR2(1000),
web_id int,

CONSTRAINT web_testimonial_fk FOREIGN KEY (web_id) REFERENCES website(web_id)
)

create table user_Order(
order_id int generated by default as identity primary key,
order_date date,
quntity int,
total_amount FLOAT,
status VARCHAR2(20),

user_id int,
CONSTRAINT user_order_fk FOREIGN KEY (user_id) REFERENCES user_login(user_id)
)

create table order_product(
order_product_id int generated by default as identity primary key,

order_id int,
meat_id int,
bread_id int,
cheese_id int,
suc_id int,


CONSTRAINT order_id_fk FOREIGN KEY (order_id) REFERENCES user_Order(order_id),
CONSTRAINT meat_id_fk FOREIGN KEY (meat_id) REFERENCES product(product_id),
CONSTRAINT bread_id_fk FOREIGN KEY (bread_id) REFERENCES product(product_id),
CONSTRAINT cheese_id_fk FOREIGN KEY (cheese_id) REFERENCES product(product_id),
CONSTRAINT suc_id_fk FOREIGN KEY (suc_id) REFERENCES product(product_id)
)

create table category_(
category_id int generated by default as identity primary key,
category_name VARCHAR2(50)

)

create table product(
product_id int generated by default as identity primary key,
product_name VARCHAR2(120),
product_price float,
product_quntity NUMBER,
is_available VARCHAR2(10),
image_path VARCHAR2(1000),


category_id int,
CONSTRAINT product_category_fk FOREIGN KEY (category_id) REFERENCES category_(category_id)
)

create table card(
card_id int generated by default as identity primary key,
card_number NUMBER,
ccv int,
expDate date,
balance float,

user_id int,
CONSTRAINT card_user_fk FOREIGN KEY (user_id) REFERENCES user_login(user_id)
)

create table cart(
cart_id int generated by default as identity primary key,

order_id int,
user_id int,
card_id int,
CONSTRAINT cart_order_id_fk FOREIGN KEY (order_id) REFERENCES user_order(order_id),
CONSTRAINT cart_user_id_fk FOREIGN KEY (user_id) REFERENCES user_login(user_id),
CONSTRAINT cart_card_id_fk FOREIGN KEY (card_id) REFERENCES card(card_id)
)
